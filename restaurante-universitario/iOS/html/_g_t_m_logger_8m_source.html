407 - (NSString *)prettyNameForFunc:(NSString *)func {
408  NSString *name = [func stringByTrimmingCharactersInSet:
409  [NSCharacterSet whitespaceAndNewlineCharacterSet]];
410  NSString *function = @"(unknown)";
411  if ([name length]) {
412  if (// Objective C __func__ and __PRETTY_FUNCTION__
413  [name hasPrefix:@"-["] || [name hasPrefix:@"+["] ||
414  // C++ __PRETTY_FUNCTION__ and other preadorned formats
415  [name hasSuffix:@")"]) {
416  function = name;
417  } else {
418  // Assume C99 __func__
419  function = [NSString stringWithFormat:@"%@()", name];
420  }
421  }
422  return function;
423 }

480 // Check the environment and the user preferences for the GTMVerboseLogging key
481 // to see if verbose logging has been enabled. The environment variable will
482 // override the defaults setting, so check the environment first.
483 // COV_NF_START
484 static BOOL IsVerboseLoggingEnabled(NSUserDefaults *userDefaults) {
485  NSString *value = [[[NSProcessInfo processInfo] environment]
486  objectForKey:kVerboseLoggingKey];
487  if (value) {
488  // Emulate [NSString boolValue] for pre-10.5
489  value = [value stringByTrimmingCharactersInSet:
490  [NSCharacterSet whitespaceAndNewlineCharacterSet]];
491  if ([[value uppercaseString] hasPrefix:@"Y"] ||
492  [[value uppercaseString] hasPrefix:@"T"] ||
493  [value intValue]) {
494  return YES;
495  } else {
496  return NO;
497  }
498  }
499  return [userDefaults boolForKey:kVerboseLoggingKey];
500 }
