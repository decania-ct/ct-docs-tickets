71 static void CheckSize(GPBCodedInputStreamState *state, size_t size) {
72  size_t newSize = state->bufferPos + size;
73  if (newSize > state->bufferSize) {
74  RaiseException(GPBCodedInputStreamErrorInvalidSize, nil);
75  }
76  if (newSize > state->currentLimit) {
77  // Fast forward to end of currentLimit;
78  state->bufferPos = state->currentLimit;
79  RaiseException(GPBCodedInputStreamErrorSubsectionLimitReached, nil);
80  }
81 }

102 static int32_t ReadRawVarint32(GPBCodedInputStreamState *state) {
103  int8_t tmp = ReadRawByte(state);
104  if (tmp >= 0) {
105  return tmp;
106  }
107  int32_t result = tmp & 0x7f;
108  if ((tmp = ReadRawByte(state)) >= 0) {
109  result |= tmp << 7;
110  } else {
111  result |= (tmp & 0x7f) << 7;
112  if ((tmp = ReadRawByte(state)) >= 0) {
113  result |= tmp << 14;
114  } else {
115  result |= (tmp & 0x7f) << 14;
116  if ((tmp = ReadRawByte(state)) >= 0) {
117  result |= tmp << 21;
118  } else {
119  result |= (tmp & 0x7f) << 21;
120  result |= (tmp = ReadRawByte(state)) << 28;
121  if (tmp < 0) {
122  // Discard upper 32 bits.
123  for (int i = 0; i < 5; i++) {
124  if (ReadRawByte(state) >= 0) {
125  return result;
126  }
127  }
128  RaiseException(GPBCodedInputStreamErrorInvalidVarInt,
129  @"Invalid VarInt32");
130  }
131  }
132  }
133  }
134  return result;
135 }

137 static int64_t ReadRawVarint64(GPBCodedInputStreamState *state) {
138  int32_t shift = 0;
139  int64_t result = 0;
140  while (shift < 64) {
141  int8_t b = ReadRawByte(state);
142  result |= (int64_t)(b & 0x7F) << shift;
143  if ((b & 0x80) == 0) {
144  return result;
145  }
146  shift += 7;
147  }
148  RaiseException(GPBCodedInputStreamErrorInvalidVarInt, @"Invalid VarInt64");
149  return 0;
150 }

226 int32_t GPBCodedInputStreamReadTag(GPBCodedInputStreamState *state) {
227  if (GPBCodedInputStreamIsAtEnd(state)) {
228  state->lastTag = 0;
229  return 0;
230  }
231 
232  state->lastTag = ReadRawVarint32(state);
233  if (state->lastTag == 0) {
234  // If we actually read zero, that's not a valid tag.
235  RaiseException(GPBCodedInputStreamErrorInvalidTag,
236  @"A zero tag on the wire is invalid.");
237  }
238  // Tags have to include a valid wireformat, check that also.
239  if (!GPBWireFormatIsValidTag(state->lastTag)) {
240  RaiseException(GPBCodedInputStreamErrorInvalidTag,
241  @"Invalid wireformat in tag.");
242  }
243  return state->lastTag;
244 }

246 NSString *GPBCodedInputStreamReadRetainedString(
247  GPBCodedInputStreamState *state) {
248  int32_t size = ReadRawVarint32(state);
249  NSString *result;
250  if (size == 0) {
251  result = @"";
252  } else {
253  CheckSize(state, size);
254  result = [[NSString alloc] initWithBytes:&state->bytes[state->bufferPos]
255  length:size
256  encoding:NSUTF8StringEncoding];
257  state->bufferPos += size;
258  if (!result) {
259 #ifdef DEBUG
260  // https://developers.google.com/protocol-buffers/docs/proto#scalar

262  @"'bytes'?");
263 #endif
264  RaiseException(GPBCodedInputStreamErrorInvalidUTF8, nil);
265  }
266  }
267  return result;

365 - (BOOL)skipField:(int32_t)tag {
366  NSAssert(GPBWireFormatIsValidTag(tag), @"Invalid tag");
367  switch (GPBWireFormatGetTagWireType(tag)) {
368  case GPBWireFormatVarint:
369  GPBCodedInputStreamReadInt32(&state_);
370  return YES;
371  case GPBWireFormatFixed64:
372  SkipRawData(&state_, sizeof(int64_t));
373  return YES;
374  case GPBWireFormatLengthDelimited:
375  SkipRawData(&state_, ReadRawVarint32(&state_));
376  return YES;
377  case GPBWireFormatStartGroup:
378  [self skipMessage];
379  GPBCodedInputStreamCheckLastTagWas(
380  &state_, GPBWireFormatMakeTag(GPBWireFormatGetTagFieldNumber(tag),
381  GPBWireFormatEndGroup));
382  return YES;
383  case GPBWireFormatEndGroup:
384  return NO;
385  case GPBWireFormatFixed32:
386  SkipRawData(&state_, sizeof(int32_t));
387  return YES;
388  }
389 }

452 - (void)readGroup:(int32_t)fieldNumber
453  message:(GPBMessage *)message
