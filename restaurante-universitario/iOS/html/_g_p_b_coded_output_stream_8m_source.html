57 // Internal helper that writes the current buffer to the output. The
58 // buffer position is reset to its initial value when this returns.
59 static void GPBRefreshBuffer(GPBOutputBufferState *state) {
60  if (state->output == nil) {
61  // We're writing to a single buffer.
62  [NSException raise:@"OutOfSpace" format:@""];
63  }
64  if (state->position != 0) {
65  NSInteger written =
66  [state->output write:state->bytes maxLength:state->position];
67  if (written != (NSInteger)state->position) {
68  [NSException raise:@"WriteFailed" format:@""];
69  }
70  state->position = 0;
71  }
72 }

82  while (YES) {

268 - (void)writeStringNoTag:(const NSString *)value {
269  size_t length = [value lengthOfBytesUsingEncoding:NSUTF8StringEncoding];
270  GPBWriteRawVarint32(&state_, (int32_t)length);
271  if (length == 0) {
272  return;
273  }
274 
275  const char *quickString =
276  CFStringGetCStringPtr((CFStringRef)value, kCFStringEncodingUTF8);
277 
278  // Fast path: Most strings are short, if the buffer already has space,
279  // add to it directly.
280  NSUInteger bufferBytesLeft = state_.size - state_.position;
281  if (bufferBytesLeft >= length) {
282  NSUInteger usedBufferLength = 0;
283  BOOL result;
284  if (quickString != NULL) {
285  memcpy(state_.bytes + state_.position, quickString, length);
286  usedBufferLength = length;
287  result = YES;
288  } else {
289  result = [value getBytes:state_.bytes + state_.position
290  maxLength:bufferBytesLeft
291  usedLength:&usedBufferLength
292  encoding:NSUTF8StringEncoding
293  options:(NSStringEncodingConversionOptions)0
294  range:NSMakeRange(0, [value length])
295  remainingRange:NULL];
296  }
297  if (result) {
298  NSAssert2((usedBufferLength == length),
299  @"Our UTF8 calc was wrong? %tu vs %zd", usedBufferLength,
300  length);
301  state_.position += usedBufferLength;
302  return;
303  }
304  } else if (quickString != NULL) {
305  [self writeRawPtr:quickString offset:0 length:length];
306  } else {
307  // Slow path: just get it as data and write it out.
308  NSData *utf8Data = [value dataUsingEncoding:NSUTF8StringEncoding];
309  NSAssert2(([utf8Data length] == length),
310  @"Strings UTF8 length was wrong? %tu vs %zd", [utf8Data length],
311  length);
312  [self writeRawData:utf8Data];
313  }
314 }

906 - (void)writeRawPtr:(const void *)value
907  offset:(size_t)offset
908  length:(size_t)length {
909  if (value == nil || length == 0) {
910  return;
911  }
912 
913  NSUInteger bufferLength = state_.size;
914  NSUInteger bufferBytesLeft = bufferLength - state_.position;
915  if (bufferBytesLeft >= length) {
916  // We have room in the current buffer.
917  memcpy(state_.bytes + state_.position, ((uint8_t *)value) + offset, length);
918  state_.position += length;
919  } else {
920  // Write extends past current buffer. Fill the rest of this buffer and
921  // flush.
922  size_t bytesWritten = bufferBytesLeft;
923  memcpy(state_.bytes + state_.position, ((uint8_t *)value) + offset,
924  bytesWritten);
925  offset += bytesWritten;
926  length -= bytesWritten;
927  state_.position = bufferLength;
928  GPBRefreshBuffer(&state_);
929  bufferLength = state_.size;
930 
931  // Now deal with the rest.
932  // Since we have an output stream, this is our buffer
933  // and buffer offset == 0
934  if (length <= bufferLength) {
935  // Fits in new buffer.
936  memcpy(state_.bytes, ((uint8_t *)value) + offset, length);
937  state_.position = length;
938  } else {
939  // Write is very big. Let's do it all at once.
940  [state_.output write:((uint8_t *)value) + offset maxLength:length];
941  }
942  }
