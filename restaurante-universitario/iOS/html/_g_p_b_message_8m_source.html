142 static NSMutableDictionary *CloneExtensionMap(NSDictionary *extensionMap,
143  NSZone *zone) {
144  if (extensionMap.count == 0) {
145  return nil;
146  }
147  NSMutableDictionary *result = [[NSMutableDictionary allocWithZone:zone]
148  initWithCapacity:extensionMap.count];
149 
150  for (GPBExtensionDescriptor *extension in extensionMap) {
151  id value = [extensionMap objectForKey:extension];
152  BOOL isMessageExtension = GPBExtensionIsMessage(extension);
153 
154  if (extension.repeated) {
155  if (isMessageExtension) {
156  NSMutableArray *list =
157  [[NSMutableArray alloc] initWithCapacity:[value count]];
158  for (GPBMessage *listValue in value) {
159  GPBMessage *copiedValue = [listValue copyWithZone:zone];
160  [list addObject:copiedValue];
161  [copiedValue release];
162  }
163  [result setObject:list forKey:extension];
164  [list release];
165  } else {
166  NSMutableArray *copiedValue = [value mutableCopyWithZone:zone];
167  [result setObject:copiedValue forKey:extension];
168  [copiedValue release];
169  }
170  } else {
171  if (isMessageExtension) {
172  GPBMessage *copiedValue = [value copyWithZone:zone];
173  [result setObject:copiedValue forKey:extension];
174  [copiedValue release];
175  } else {
176  [result setObject:value forKey:extension];
177  }
178  }
179  }
180 
181  return result;
182 }

912 - (void)copyFieldsInto:(GPBMessage *)message
913  zone:(NSZone *)zone
914  descriptor:(GPBDescriptor *)descriptor {
915  // Copy all the storage...
916  memcpy(message->messageStorage_, messageStorage_, descriptor->storageSize_);
917 
918  GPBFileSyntax syntax = descriptor.file.syntax;
919 
920  // Loop over the fields doing fixup...
921  for (GPBFieldDescriptor *field in descriptor->fields_) {
922  if (GPBFieldIsMapOrArray(field)) {
923  id value = GPBGetObjectIvarWithFieldNoAutocreate(self, field);
924  if (value) {
925  // We need to copy the array/map, but the catch is for message fields,
926  // we also need to ensure all the messages as those need copying also.
927  id newValue;
928  if (GPBFieldDataTypeIsMessage(field)) {
929  if (field.fieldType == GPBFieldTypeRepeated) {
930  NSArray *existingArray = (NSArray *)value;
931  NSMutableArray *newArray =
932  [[NSMutableArray alloc] initWithCapacity:existingArray.count];
933  newValue = newArray;
934  for (GPBMessage *msg in existingArray) {
935  GPBMessage *copiedMsg = [msg copyWithZone:zone];
936  [newArray addObject:copiedMsg];
937  [copiedMsg release];
938  }
939  } else {
940  if (field.mapKeyDataType == GPBDataTypeString) {
941  // Map is an NSDictionary.
942  NSDictionary *existingDict = value;
943  NSMutableDictionary *newDict = [[NSMutableDictionary alloc]
944  initWithCapacity:existingDict.count];
945  newValue = newDict;
946  [existingDict enumerateKeysAndObjectsUsingBlock:^(NSString *key,
947  GPBMessage *msg,
948  BOOL *stop) {
949 #pragma unused(stop)
950  GPBMessage *copiedMsg = [msg copyWithZone:zone];
951  [newDict setObject:copiedMsg forKey:key];
952  [copiedMsg release];
953  }];
954  } else {
955  // Is one of the GPB*ObjectDictionary classes. Type doesn't
956  // matter, just need one to invoke the selector.
957  GPBInt32ObjectDictionary *existingDict = value;
958  newValue = [existingDict deepCopyWithZone:zone];
959  }
960  }
961  } else {
962  // Not messages (but is a map/array)...
963  if (field.fieldType == GPBFieldTypeRepeated) {
964  if (GPBFieldDataTypeIsObject(field)) {
965  // NSArray
966  newValue = [value mutableCopyWithZone:zone];
967  } else {
968  // GPB*Array
969  newValue = [value copyWithZone:zone];
970  }
971  } else {
972  if (field.mapKeyDataType == GPBDataTypeString) {
973  // NSDictionary
974  newValue = [value mutableCopyWithZone:zone];
975  } else {
976  // Is one of the GPB*Dictionary classes. Type doesn't matter,
977  // just need one to invoke the selector.
978  GPBInt32Int32Dictionary *existingDict = value;
979  newValue = [existingDict copyWithZone:zone];
980  }
981  }
982  }
983  // We retain here because the memcpy picked up the pointer value and
984  // the next call to SetRetainedObject... will release the current value.
985  [value retain];
986  GPBSetRetainedObjectIvarWithFieldInternal(message, field, newValue,
987  syntax);
988  }
989  } else if (GPBFieldDataTypeIsMessage(field)) {
990  // For object types, if we have a value, copy it. If we don't,
991  // zero it to remove the pointer to something that was autocreated
992  // (and the ptr just got memcpyed).
993  if (GPBGetHasIvarField(self, field)) {
994  GPBMessage *value = GPBGetObjectIvarWithFieldNoAutocreate(self, field);
995  GPBMessage *newValue = [value copyWithZone:zone];
996  // We retain here because the memcpy picked up the pointer value and
997  // the next call to SetRetainedObject... will release the current value.
998  [value retain];
999  GPBSetRetainedObjectIvarWithFieldInternal(message, field, newValue,
1000  syntax);
1001  } else {
1002  uint8_t *storage = (uint8_t *)message->messageStorage_;
1003  id *typePtr = (id *)&storage[field->description_->offset];
1004  *typePtr = NULL;
1005  }
1006  } else if (GPBFieldDataTypeIsObject(field) &&
1007  GPBGetHasIvarField(self, field)) {
1008  // A set string/data value (message picked off above), copy it.
1009  id value = GPBGetObjectIvarWithFieldNoAutocreate(self, field);
1010  id newValue = [value copyWithZone:zone];
1011  // We retain here because the memcpy picked up the pointer value and
1012  // the next call to SetRetainedObject... will release the current value.
1013  [value retain];
1014  GPBSetRetainedObjectIvarWithFieldInternal(message, field, newValue,
1015  syntax);
1016  } else {
1017  // memcpy took care of the rest of the primitive fields if they were set.
1018  }
1019  } // for (field in descriptor->fields_)
1020 }

2627  int32_t hasIndex = GPBFieldHasIndex(field);

2633  }

2639  GPBDataType fieldDataType = GPBGetFieldDataType(field);

2642  case GPBDataTypeBool: {

2646  // matter since the offset is never negative)

2714  }
2715  }

2720 // It is very difficult to implement a generic hash for ProtoBuf messages that
2721 // will perform well. If you need hashing on your ProtoBufs (eg you are using
2722 // them as dictionary keys) you will probably want to implement a ProtoBuf
2723 // message specific hash as a category on your protobuf class. Do not make it a
2724 // category on GPBMessage as you will conflict with this hash, and will possibly
2725 // override hash for all generated protobufs. A good implementation of hash will
2726 // be really fast, so we would recommend only hashing protobufs that have an
2727 // identifier field of some kind that you can easily hash. If you implement
2728 // hash, we would strongly recommend overriding isEqual: in your category as
2729 // well, as the default implementation of isEqual: is extremely slow, and may
2730 // drastically affect performance in large sets.
2731 - (NSUInteger)hash {
2732  GPBDescriptor *descriptor = [[self class] descriptor];
2733  const NSUInteger prime = 19;
2734  uint8_t *storage = (uint8_t *)messageStorage_;
2735 
2736  // Start with the descriptor and then mix it with some instance info.
2737  // Hopefully that will give a spread based on classes and what fields are set.
2738  NSUInteger result = (NSUInteger)descriptor;
2739 
2740  for (GPBFieldDescriptor *field in descriptor->fields_) {
2741  if (GPBFieldIsMapOrArray(field)) {
2742  // Exact type doesn't matter, just check if there are any elements.
2743  NSArray *mapOrArray = GPBGetObjectIvarWithFieldNoAutocreate(self, field);
2744  NSUInteger count = mapOrArray.count;
2745  if (count) {
2746  // NSArray/NSDictionary use count, use the field number and the count.
2747  result = prime * result + GPBFieldNumber(field);
2748  result = prime * result + count;
2749  }
2750  } else if (GPBGetHasIvarField(self, field)) {
2751  // Just using the field number seemed simple/fast, but then a small
2752  // message class where all the same fields are always set (to different
2753  // things would end up all with the same hash, so pull in some data).
2754  GPBDataType fieldDataType = GPBGetFieldDataType(field);
2755  size_t fieldOffset = field->description_->offset;
2756  switch (fieldDataType) {
2757  case GPBDataTypeBool: {
2758  // Bools are stored in has_bits to avoid needing explicit space in
2759  // the storage structure.
2760  // (the field number passed to the HasIvar helper doesn't really
2761  // matter since the offset is never negative)
2762  BOOL value = GPBGetHasIvar(self, (int32_t)(fieldOffset), 0);
2763  result = prime * result + value;
2764  break;
2765  }
2766  case GPBDataTypeSFixed32:
2767  case GPBDataTypeInt32:
2768  case GPBDataTypeSInt32:
2769  case GPBDataTypeEnum:
2770  case GPBDataTypeFixed32:
2771  case GPBDataTypeUInt32:
2772  case GPBDataTypeFloat: {
2773  GPBInternalCompileAssert(sizeof(float) == sizeof(uint32_t), float_not_32_bits);
2774  // These are all 32bit, just mix it in.
2775  uint32_t *valPtr = (uint32_t *)&storage[fieldOffset];
2776  result = prime * result + *valPtr;
2777  break;
2778  }
2779  case GPBDataTypeSFixed64:
2780  case GPBDataTypeInt64:
2781  case GPBDataTypeSInt64:
2782  case GPBDataTypeFixed64:
2783  case GPBDataTypeUInt64:
2784  case GPBDataTypeDouble: {
2785  GPBInternalCompileAssert(sizeof(double) == sizeof(uint64_t), double_not_64_bits);
2786  // These are all 64bit, just mix what fits into an NSUInteger in.
2787  uint64_t *valPtr = (uint64_t *)&storage[fieldOffset];
2788  result = prime * result + (NSUInteger)(*valPtr);
2789  break;
2790  }
2791  case GPBDataTypeBytes:
2792  case GPBDataTypeString: {
2793  // Type doesn't matter here, they both implement -hash:.
2794  id *valPtr = (id *)&storage[fieldOffset];
2795  result = prime * result + [*valPtr hash];
2796  break;
2797  }
2798 
2799  case GPBDataTypeMessage:
2800  case GPBDataTypeGroup: {
2801  GPBMessage **valPtr = (GPBMessage **)&storage[fieldOffset];
2802  // Could call -hash on the sub message, but that could recurse pretty
2803  // deep; follow the lead of NSArray/NSDictionary and don't really
2804  // recurse for hash, instead use the field number and the descriptor
2805  // of the sub message. Yes, this could suck for a bunch of messages
2806  // where they all only differ in the sub messages, but if you are
2807  // using a message with sub messages for something that needs -hash,
2808  // odds are you are also copying them as keys, and that deep copy
2809  // will also suck.
2810  result = prime * result + GPBFieldNumber(field);
2811  result = prime * result + (NSUInteger)[[*valPtr class] descriptor];
2812  break;
2813  }
2814  } // switch()
2815  }
2816  }
2817 
2818  // Unknowns and extensions are not included.
2819 
2820  return result;
2821 }
