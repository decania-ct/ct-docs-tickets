49 + (NSData *)gtm_dataByCompressingBytes:(const void *)bytes
50  length:(NSUInteger)length
51  compressionLevel:(int)level
52  mode:(CompressionMode)mode
53  error:(NSError **)error {
54  if (!bytes || !length) {
55  return nil;
56  }
57 
58 #if defined(__LP64__) && __LP64__
59  // Don't support > 32bit length for 64 bit, see note in header.
60  if (length > UINT_MAX) {
61  if (error) {
62  *error = [NSError errorWithDomain:GTMNSDataZlibErrorDomain
63  code:GTMNSDataZlibErrorGreaterThan32BitsToCompress
64  userInfo:nil];
65  }
66  return nil;
67  }
68 #endif
69 
70  if (level == Z_DEFAULT_COMPRESSION) {
71  // the default value is actually outside the range, so we have to let it
72  // through specifically.
73  } else if (level < Z_BEST_SPEED) {
74  level = Z_BEST_SPEED;
75  } else if (level > Z_BEST_COMPRESSION) {
76  level = Z_BEST_COMPRESSION;
77  }
78 
79  z_stream strm;
80  bzero(&strm, sizeof(z_stream));
81 
82  int memLevel = 8; // the default
83  int windowBits = 15; // the default
84  switch (mode) {
85  case CompressionModeZlib:
86  // nothing to do
87  break;
88 
89  case CompressionModeGzip:
90  windowBits += 16; // enable gzip header instead of zlib header
91  break;
92 
93  case CompressionModeRaw:
94  windowBits *= -1; // Negative to mean no header.
95  break;
96  }
97  int retCode;
98  if ((retCode = deflateInit2(&strm, level, Z_DEFLATED, windowBits,
99  memLevel, Z_DEFAULT_STRATEGY)) != Z_OK) {
100  // COV_NF_START - no real way to force this in a unittest (we guard all args)
101  if (error) {
102  NSDictionary *userInfo = [NSDictionary dictionaryWithObject:[NSNumber numberWithInt:retCode]
103  forKey:GTMNSDataZlibErrorKey];
104  *error = [NSError errorWithDomain:GTMNSDataZlibErrorDomain
105  code:GTMNSDataZlibErrorInternal
106  userInfo:userInfo];
107  }
108  return nil;
109  // COV_NF_END
110  }
111 
112  // hint the size at 1/4 the input size
113  NSMutableData *result = [NSMutableData dataWithCapacity:(length/4)];
114  unsigned char output[kChunkSize];
115 
116  // setup the input
117  strm.avail_in = (unsigned int)length;
118  strm.next_in = (unsigned char*)bytes;
119 
120  // loop to collect the data
121  do {
122  // update what we're passing in
123  strm.avail_out = kChunkSize;
124  strm.next_out = output;
125  retCode = deflate(&strm, Z_FINISH);
126  if ((retCode != Z_OK) && (retCode != Z_STREAM_END)) {
127  // COV_NF_START - no real way to force this in a unittest
128  // (in inflate, we can feed bogus/truncated data to test, but an error
129  // here would be some internal issue w/in zlib, and there isn't any real
130  // way to test it)
131  if (error) {
132  NSDictionary *userInfo = [NSDictionary dictionaryWithObject:[NSNumber numberWithInt:retCode]
133  forKey:GTMNSDataZlibErrorKey];
134  *error = [NSError errorWithDomain:GTMNSDataZlibErrorDomain
135  code:GTMNSDataZlibErrorInternal
136  userInfo:userInfo];
137  }
138  deflateEnd(&strm);
139  return nil;
140  // COV_NF_END
141  }
142  // collect what we got
143  unsigned gotBack = kChunkSize - strm.avail_out;
144  if (gotBack > 0) {
145  [result appendBytes:output length:gotBack];
146  }
147 
148  } while (retCode == Z_OK);
149 
150  // if the loop exits, we used all input and the stream ended
151  _GTMDevAssert(strm.avail_in == 0,
152  @"thought we finished deflate w/o using all input, %u bytes left",
153  strm.avail_in);
154  _GTMDevAssert(retCode == Z_STREAM_END,
155  @"thought we finished deflate w/o getting a result of stream end, code %d",
156  retCode);
157 
158  // clean up
159  deflateEnd(&strm);
160 
161  return result;
162 } // gtm_dataByCompressingBytes:length:compressionLevel:useGzip:

164 + (NSData *)gtm_dataByInflatingBytes:(const void *)bytes
165  length:(NSUInteger)length
166  isRawData:(BOOL)isRawData
167  error:(NSError **)error {
168  if (!bytes || !length) {
169  return nil;
170  }
171 
172 #if defined(__LP64__) && __LP64__
173  // Don't support > 32bit length for 64 bit, see note in header.
174  if (length > UINT_MAX) {
175  return nil;
176  }
177 #endif
178 
179  z_stream strm;
180  bzero(&strm, sizeof(z_stream));
181 
182  // setup the input
183  strm.avail_in = (unsigned int)length;
184  strm.next_in = (unsigned char*)bytes;
185 
186  int windowBits = 15; // 15 to enable any window size
187  if (isRawData) {
188  windowBits *= -1; // make it negative to signal no header.
189  } else {
190  windowBits += 32; // and +32 to enable zlib or gzip header detection.
191  }
192 
193  int retCode;
194  if ((retCode = inflateInit2(&strm, windowBits)) != Z_OK) {
195  // COV_NF_START - no real way to force this in a unittest (we guard all args)
196  if (error) {
197  NSDictionary *userInfo = [NSDictionary dictionaryWithObject:[NSNumber numberWithInt:retCode]
198  forKey:GTMNSDataZlibErrorKey];
199  *error = [NSError errorWithDomain:GTMNSDataZlibErrorDomain
200  code:GTMNSDataZlibErrorInternal
201  userInfo:userInfo];
202  }
203  return nil;
204  // COV_NF_END
205  }
206 
207  // hint the size at 4x the input size
208  NSMutableData *result = [NSMutableData dataWithCapacity:(length*4)];
209  unsigned char output[kChunkSize];
210 
211  // loop to collect the data
212  do {
213  // update what we're passing in
214  strm.avail_out = kChunkSize;
215  strm.next_out = output;
216  retCode = inflate(&strm, Z_NO_FLUSH);
217  if ((retCode != Z_OK) && (retCode != Z_STREAM_END)) {
218  if (error) {
219  NSMutableDictionary *userInfo =
220  [NSMutableDictionary dictionaryWithObject:[NSNumber numberWithInt:retCode]
221  forKey:GTMNSDataZlibErrorKey];
222  if (strm.msg) {
223  NSString *message = [NSString stringWithUTF8String:strm.msg];
224  if (message) {
225  [userInfo setObject:message forKey:NSLocalizedDescriptionKey];
226  }
227  }
228  *error = [NSError errorWithDomain:GTMNSDataZlibErrorDomain
229  code:GTMNSDataZlibErrorInternal
230  userInfo:userInfo];
231  }
232  inflateEnd(&strm);
233  return nil;
234  }
235  // collect what we got
236  unsigned gotBack = kChunkSize - strm.avail_out;
237  if (gotBack > 0) {
238  [result appendBytes:output length:gotBack];
239  }
240 
241  } while (retCode == Z_OK);
242 
243  // make sure there wasn't more data tacked onto the end of a valid compressed
244  // stream.
245  if (strm.avail_in != 0) {
246  if (error) {
247  NSDictionary *userInfo =
248  [NSDictionary dictionaryWithObject:[NSNumber numberWithUnsignedInt:strm.avail_in]
249  forKey:GTMNSDataZlibRemainingBytesKey];
250  *error = [NSError errorWithDomain:GTMNSDataZlibErrorDomain
251  code:GTMNSDataZlibErrorDataRemaining
252  userInfo:userInfo];
253  }
254  result = nil;
255  }
256  // the only way out of the loop was by hitting the end of the stream
257  _GTMDevAssert(retCode == Z_STREAM_END,
258  @"thought we finished inflate w/o getting a result of stream end, code %d",
259  retCode);
260 
261  // clean up
262  inflateEnd(&strm);
263 
264  return result;
265 } // gtm_dataByInflatingBytes:length:windowBits:
